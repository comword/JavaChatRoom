388  public static String getHexdump(ByteBuffer in) {
389  int size = in.remaining();
390  if (size == 0) {
391  return "empty";
392  }
393  StringBuilder out = new StringBuilder((in.remaining() * 3) - 1);
394  int mark = in.position();
395  int byteValue = in.get() & 0xFF;
396  out.append((char) highDigits[byteValue]);
397  out.append((char) lowDigits[byteValue]);
398  size--;
399  for (; size > 0; size--) {
400  out.append(' ');
401  byteValue = in.get() & 0xFF;
402  out.append((char) highDigits[byteValue]);
403  out.append((char) lowDigits[byteValue]);
404  }
405  in.position(mark);
406  return out.toString();
