133  private void skipField(byte type) {
134  switch (type) {
135  case TarsStructBase.BYTE:
136  skip(1);
137  break;
138  case TarsStructBase.SHORT:
139  skip(2);
140  break;
141  case TarsStructBase.INT:
142  skip(4);
143  break;
144  case TarsStructBase.LONG:
145  skip(8);
146  break;
147  case TarsStructBase.FLOAT:
148  skip(4);
149  break;
150  case TarsStructBase.DOUBLE:
151  skip(8);
152  break;
153  case TarsStructBase.STRING1: {
154  int len = bs.get();
155  if (len < 0) len += 256;
156  skip(len);
157  break;
158  }
159  case TarsStructBase.STRING4: {
160  skip(bs.getInt());
161  break;
162  }
163  case TarsStructBase.MAP: {
164  int size = read(0, 0, true);
165  for (int i = 0; i < size * 2; ++i)
166  skipField();
167  break;
168  }
169  case TarsStructBase.LIST: {
170  int size = read(0, 0, true);
171  for (int i = 0; i < size; ++i)
172  skipField();
173  break;
174  }
175  case TarsStructBase.SIMPLE_LIST: {
176  HeadData hd = new HeadData();
177  readHead(hd);
178  if (hd.type != TarsStructBase.BYTE) {
179  throw new TarsDecodeException("skipField with invalid type, type value: " + type + ", " + hd.type);
180  }
181  int size = read(0, 0, true);
182  skip(size);
183  break;
184  }
185  case TarsStructBase.STRUCT_BEGIN:
186  skipToStructEnd();
187  break;
188  case TarsStructBase.STRUCT_END:
189  case TarsStructBase.ZERO_TAG:
190  break;
191  default:
192  throw new TarsDecodeException("invalid type.");
193  }
194  }
